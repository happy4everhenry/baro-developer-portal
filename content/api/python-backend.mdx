## Running the Server

```python filename="API Testing with cURL"

# Health check
$ curl http://localhost:8000/health

$ curl "http://localhost:8000/api/market/indices?ids=kospi,nasdaq"
```

REST API client for Korean stock data. Handles OAuth headers, sign-based price direction, Korean number formatting. Supports quote, chart (daily/weekly/monthly), orderbook (10 levels), and 5 ranking types (market cap, volume, popular, gainers, losers).

## Yahoo Finance Integration

```python filename="market_service.py"
def _fetch_yahoo_chart(symbol: str) -> dict | None:
    """Fetch chart data from Yahoo Finance v8 API."""
    url = f"https://query1.finance.yahoo.com/v8/finance/chart/{symbol}"
    params = {"range": "2d", "interval": "1d"}
    headers = {"User-Agent": "Mozilla/5.0"}
    response = requests.get(url, params=params, headers=headers, timeout=10)
    return response.json()["chart"]["result"][0]

def _parse_price_data(chart: dict) -> tuple[float, float] | None:
    """Extract (current_price, previous_close) from chart metadata."""
    meta = chart["meta"]
    current = meta.get("regularMarketPrice")
    prev = meta.get("chartPreviousClose") or meta.get("previousClose")
    return (current, prev) if current and prev else None

def _get_market_status(chart: dict) -> str:
    """Map Yahoo market state to Korean display string."""
    state = chart["meta"].get("marketState", "")
    return {
        "REGULAR": "장중", "OPEN": "장중",
        "PRE": "장전", "PREPRE": "장전",
        "POST": "장후", "POSTPOST": "장후",
    }.get(state, "장종료")
```
